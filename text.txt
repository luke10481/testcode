    function GetUrlParam() {
        let url = document.location.toString();
        url = decodeURIComponent(url);
        let arrObj = url.split("?");
        let params = Object.create(null)

        if (arrObj.length > 1) {
            arrObj = arrObj[1].replace('#', '&').split('&');
            arrObj.forEach(item => {
                item = item.split("=");
                params[item[0]] = item[1]
            })
        }
        return params;
    }
